# crontab 설정 — `crontab -e`로 열고 아래 내용 추가 (ubuntu 사용자)
#
# ⚠️ 서버 시간은 UTC다. 한국시간(KST) = UTC + 9.
#    아침 시간대 잡은 UTC로 보면 "전날"이 되므로 요일(day-of-week)도 하루 당겨야 한다.
#    예) KST 월~금 08:00  ==  UTC 일~목(0-4) 23:00
#        KST 월~금 07:00  ==  UTC 일~목(0-4) 22:00
#    cron 요일: 0=일 1=월 2=화 3=수 4=목 5=금 6=토
#    (이 함정 때문에 카카오 브리핑이 KST 월요일에 안 오던 버그가 있었음 → 0-4로 수정)

# 1) 가격 알림 체크 — 미국 장중 30분마다 (KST 23:30~06:00 = UTC 14:30~21:00, 평일)
*/30 14-21 * * 1-5 cd /home/ubuntu/stock-agent && .venv/bin/python check_alerts.py >> logs/alerts.log 2>&1

# 1-b) 시장 데이터 사전 수집 — 매일 KST 06:15 (UTC 21:15, 미 증시 마감 후)
#       보유/관심/추적 종목 + 지수를 미리 캐시(data/cache/) → 페이지·시그널·데일리신문이 즉시 읽음.
#       데일리신문(UTC 22:00)·카카오(UTC 23:00)보다 먼저 돌아야 신선한 캐시를 읽으므로 순서 주의.
15 21 * * * cd /home/ubuntu/stock-agent && .venv/bin/python scripts/collect_market_data.py >> logs/collect_market.log 2>&1

# 2) 영상 알림 갱신 — 매일 KST 06:00 (UTC 21:00)
0 21 * * * cd /home/ubuntu/stock-agent && .venv/bin/python scripts/refresh_video_alerts.py >> logs/video_alerts.log 2>&1

# 3) 데일리 신문 자동 발행 — 매일 KST 07:00 (UTC 22:00)
0 22 * * * cd /home/ubuntu/stock-agent && .venv/bin/python scripts/auto_publish_paper.py >> logs/auto_paper.log 2>&1

# 4) 카카오 데일리 브리핑 — KST 월~금 08:00 (UTC 일~목 23:00 → 요일 0-4 주의!)
0 23 * * 0-4 cd /home/ubuntu/stock-agent && STOCK_AGENT_USER=admin .venv/bin/python scripts/daily_kakao_briefing.py >> logs/kakao_briefing.log 2>&1

# 로그 정리 (매주 일요일, 30일 이상된 로그 삭제)
0 0 * * 0 find /home/ubuntu/stock-agent/logs -name "*.log" -mtime +30 -delete
